562 if ((*q++ = *key << 1))
563 key++;
564 }
565 if (des_setkey((unsigned char *)keybuf))
566 return (NULL((void*)0));
567
568 if (*setting == _PASSWORD_EFMT1'_')
569 {
570 /*
571 * "new"-style:
572 * setting - underscore, 4 bytes of count, 4 bytes of salt
573 * key - unlimited characters
574 */
575 for (i = 1, count = 0; i < 5; i++)
576 count |= ascii_to_bin(setting[i]) << (i - 1) * 6;
577
578 for (i = 5, salt = 0; i < 9; i++)
579 salt |= ascii_to_bin(setting[i]) << (i - 5) * 6;
580
581 while (*key)
582 {
583 /*
584 * Encrypt the key with itself.
585 */
586 if (des_cipher((unsigned char *)keybuf, (unsigned char *)keybuf, 0,
587 1))
588 return (NULL((void*)0));
589 /*
590 * And XOR with the next 8 characters of the key.
591 */
592 q = (unsigned char *)keybuf;
593 while (((q - (unsigned char *)keybuf) < sizeof(keybuf)) && *key)
594 *q++ ^= *key++ << 1;
595
596 if (des_setkey((unsigned char *)keybuf))
597 return (NULL((void*)0));
598 }
599 strncpy((char *)output, setting, 9);
600
601 /*
602 * Double check that we weren't given a short setting.
603 * If we were, the above code will probably have created
604 * wierd values for count and salt, but we don't really care.
605 * Just make sure the output string doesn't have an extra
606 * NUL in it.
607 */
608 output[9] = '\0';
609 p = output + strlen((const char *)output);
610 }
611 else
612 {
613 /*
614 * "old"-style:
615 * setting - 2 bytes of salt
616 * key - up to 8 characters
617 */
618 count = 25;
619
620 salt = (ascii_to_bin(setting[1]) << 6) | ascii_to_bin(setting[0]);
621
622 output[0] = setting[0];
623 /*
624 * If the encrypted password that the salt was extracted from
625 * is only 1 character long, the salt will be corrupted. We
626 * need to ensure that the output string doesn't have an extra
627 * NUL in it!
628 */
629 output[1] = setting[1] ? setting[1] : output[0];
630
631 p = output + 2;
632 }
633 setup_salt(salt);
634 /*
635 * Do it.
636 */
637 if (do_des(0, 0, &r0, &r1, count))
638 return (NULL((void*)0));
639 /*
640 * Now encode the result...
641 */
642 l = (r0 >> 8);
643 *p++ = ascii64[(l >> 18) & 0x3f];
644 *p++ = ascii64[(l >> 12) & 0x3f];
645 *p++ = ascii64[(l >> 6) & 0x3f];
646 *p++ = ascii64[l & 0x3f];
647
648 l = (r0 << 16) | ((r1 >> 16) & 0xffff);
649 *p++ = ascii64[(l >> 18) & 0x3f];
650 *p++ = ascii64[(l >> 12) & 0x3f];
651 *p++ = ascii64[(l >> 6) & 0x3f];
652 *p++ = ascii64[l & 0x3f];
653
654 l = r1 << 2;
655 *p++ = ascii64[(l >> 12) & 0x3f];
656 *p++ = ascii64[(l >> 6) & 0x3f];
657 *p++ = ascii64[l & 0x3f];
658 *p = 0;
659
660 return ((char *)output);
661}
