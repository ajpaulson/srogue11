137bool_Bool dosave()
138{

140
141 ignore();
142 setuid(playuid);
143 setgid(playgid);
144 umask(022);
145
146 if (file_name[0] != '\0') {
147 if ((savef = fopen(file_name, "w")) != NULL((void*)0)) {
148 save_file(savef);
149 return TRUE1;
150 }
151 }
152 return FALSE0;
153}
